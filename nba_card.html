async function loadData() { try { const urlParams = new URLSearchParams(window.location.search); let targetDate = urlParams.get('date'); // TIMEZONE ALIGNMENT: Ensures the browser perfectly matches the Python bot's "today" if (!targetDate) { const estDate = new Date(new Date().toLocaleString("en-US", {timeZone: "America/New_York"})); const yyyy = estDate.getFullYear(); const mm = String(estDate.getMonth() + 1).padStart(2, '0'); const dd = String(estDate.getDate()).padStart(2, '0'); targetDate = `${yyyy}-${mm}-${dd}`; } document.getElementById('date-indicator').value = targetDate; const [nbaRes, playersRes] = await Promise.all([ fetch(`https://nbastartingfive.com/data/${targetDate}.json?v=${new Date().getTime()}`), fetch('data/players.json') ]); if (nbaRes.ok) { const data = await nbaRes.json(); // STRICT FILTER RESTORED: Only allows games into the dropdown if AT LEAST ONE // team has 5 verified starters. allGames = data.games.filter(g => { if (g.date !== targetDate) return false; if (!g.teams || !g.rosters) return false; const t1 = g.teams[0]; const t2 = g.teams[1]; const hasRoster1 = g.rosters[t1] && g.rosters[t1].players && g.rosters[t1].players.length >= 5; const hasRoster2 = g.rosters[t2] && g.rosters[t2].players && g.rosters[t2].players.length >= 5; if (!hasRoster1 && !hasRoster2) return false; // Check the Python verified flag const t1Verified = hasRoster1 && g.rosters[t1].players.slice(0, 5).every(p => p.verified === true); const t2Verified = hasRoster2 && g.rosters[t2].players.slice(0, 5).every(p => p.verified === true); return t1Verified || t2Verified; }); } if (playersRes.ok) { playerDatabase = await playersRes.json(); console.log("✅ Player Database loaded successfully!"); } } catch (e) { console.error("Failed to load data.", e); } populateDropdown(); const urlParams = new URLSearchParams(window.location.search); const botTeam = urlParams.get('team'); const botSide = urlParams.get('side'); // Find game by Team Abbreviation if (botTeam !== null) { const foundIndex = allGames.findIndex(g => g.teams.includes(botTeam)); if (foundIndex !== -1) { document.getElementById('game-selector').value = foundIndex; } } if (botSide !== null) { document.getElementById('team-selector').value = botSide; } if (allGames.length > 0) { renderCard(); } else { document.getElementById('team-name').textContent = "NO LINEUPS YET"; document.getElementById('team-name').setAttribute('data-text', "NO LINEUPS YET"); document.getElementById('matchup-info').textContent = "Check back closer to tipoff."; } }
